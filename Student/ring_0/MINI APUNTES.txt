///////////////////////////// Arrays Bidimensionales -> MALLOC



#include <stdio.h>
#include <stdlib.h>

char **func(void)
{
    size_t  row = 3;
    size_t  col = 3;
    char  count;
    
    // Creo las Filas
    char **frases = (char**)malloc(row * sizeof(char*));
    
    // Creo las Columnas
    for(size_t i=0; i<row; i++)
        frases[i] = (char*)malloc(col * sizeof(char));
        
    // Relleno con datos
    count = 'a';
    for (size_t i = 0; i < row; i++)
        for (size_t j = 0; j < col; j++)
            frases[i][j] = count++; // OR *(*(arr+i)+j)
        
    return (frases);
}
int main()
{
    char **frases = func();
    printf("%s", frases[1]);
    return 0;
}
